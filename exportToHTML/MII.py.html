37   def CreaRegla(IPORIGEN, IPDESTINO, PROTO, PUERTODESTINO): 
38       handler.log.info('creando regla: ' + IPORIGEN + ':' + str(PUERTODESTINO) + ' -> ' + IPDESTINO + ':' + str(PUERTODESTINO) + ' ' + PROTO) 
39       try: 
40           EXEC='iptables -t ' + TABLENAME + ' -R ' + PRECHAIN + ' 1 -p ' + PROTO + ' -s 0/0 -d ' + IPSBC + ' --dport ' + str(PUERTODESTINO) + ' -j DNAT --to ' + IPDESTINO + ':' + str(PUERTODESTINO) 
41           STATUS = execute(EXEC) 
42           handler.log.debug('ejecutado: %s STATUS (%s)', EXEC, STATUS) 
43            
44           EXEC='iptables -t ' + TABLENAME + ' -R ' + POSTCHAIN + ' 1 -o ' + IFACE + ' -d ' + IPDESTINO + ' -j SNAT --to-source ' + IPSBC 
45           STATUS = execute(EXEC) 
46           handler.log.debug('ejecutado: %s STATUS (%s)', EXEC, STATUS) 
47            
48           handler.log.info('regla aplicada correctamente') 
49       except Exception as message: 
50               handler.log.error('no se pudo crear la regla') 
51               handler.log.exception(message)

124  def ModificaReglas(): 
125      obtieneMejorServidor = 1 
126      while obtieneMejorServidor: 
127          try: 
128              ReglasEspeciales() 
129              handler.log.info('obteniendo mejor servidor para reglas de IPTABLES') 
130              DESTINO = MIS.ConsultaMejorServidor() 
131              handler.log.debug('mejor servidor: ' + str(DESTINO)) 
132              #LimpiaReglas() 
133              if str(DESTINO) == 'None': 
134                  handler.log.info('modificando reglas de IPTABLES con servidor por defecto') 
135                  ReglaPorDefecto() 
136              else: 
137                  handler.log.info('modificando reglas de IPTABLES con mejor servidor %s', DESTINO) 
138                  CreaRegla(IPSBC, DESTINO, PROTOCOLODEFECTO, PUERTODEFECTO) 
139              handler.log.info('aplicando reglas especiales en IPTABLES') 
140              #ReglasEspeciales() 
141          except Exception as message: 
142              handler.log.error('error al aplicar la reglas en IPTABLES') 
143              handler.log.exception(message) 
144          finally: 
145              #MuestraReglas() 
146              sleep(float(SLEEPREGLAMEJORSERVIDOR))
