91   def AgregaServidor(FQDN, PUERTO): 
92       try: 
93           handler.log.debug('agregando servidor: ' + FQDN + ':' + str(PUERTO)) 
94           conexion = connect(SBCDB, isolation_level=None) 
95           cursor=conexion.cursor() 
96           PUERTODB = cursor.execute('SELECT puerto FROM ' + TABLA_SERVIDOR + ' WHERE fqdn = ?;', ([FQDN])).fetchall() 
97           MODIFICADO = time() 
98           # si el servidor ya existe en la base de datos 
99           if PUERTODB: 
100              # pero tiene distinto puerto al registrado 
101              if PUERTO is not str(PUERTODB[0][0]): 
102                  handler.log.debug('actualizando puerto de servidor existente, de ' + str(PUERTODB[0][0]) + ' a ' + PUERTO) 
103                  cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET puerto = ? WHERE fqdn = ?;', (PUERTO, FQDN)) 
104                  cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET modificado = ? WHERE fqdn = ?;', (MODIFICADO, FQDN)) 
105              cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET activo = ? WHERE fqdn = ?;', ('TRUE', FQDN)) 
106              cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET intento = ? WHERE fqdn = ?;', (0, FQDN)) 
107          # si el servidor no existe 
108          else: 
109              handler.log.debug('agregando servidor nuevo: ' + FQDN + ':' + str(PUERTO)) 
110              cursor.execute('INSERT INTO ' + TABLA_SERVIDOR + ' (fqdn, puerto, activo, modificado, intento) VALUES (?, ?, ?, ?, ?);', (FQDN, PUERTO, 'TRUE', MODIFICADO, 0)) 
111      except Exception as message: 
112          handler.log.error('no se puede agregar servidor: %s', message) 
113          handler.log.exception(message) 
114          exit(1)

191  def ServidorConProblemas(FQDN, PUERTO): 
192      try: 
193          handler.log.debug('servidor con problemas: ' + FQDN + ':' + str(PUERTO)) 
194          conexion = connect(SBCDB, isolation_level=None) 
195          cursor=conexion.cursor() 
196          INTENTO = cursor.execute('SELECT intento FROM ' + TABLA_SERVIDOR + ' WHERE fqdn = ?;', ([FQDN])).fetchall() 
197          MODIFICADO = time() 
198          if INTENTO: 
199              # si se ha intentado menos de NUMEROINTENTOS veces consultar el servidor 
200              if INTENTO < INTENTOSSERVERDOWN: 
201                  handler.log.debug('actualizando intentos con problemas de servidor a ' + (INTENTO + 1)) 
202                  cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET intento = ? WHERE fqdn = ?;', ((INTENTO + 1), FQDN)) 
203                  cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET modificado = ? WHERE fqdn = ?;', (MODIFICADO, FQDN)) 
204              # si se ha intentado menos de NUMEROINTENTOS veces consultar el servidor 
205              else: 
206                  handler.log.debug('actualizando a inactivo estado de servidor') 
207                  cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET activo = ? WHERE fqdn = ?;', ('FALSE', FQDN)) 
208                  cursor.execute('UPDATE ' + TABLA_SERVIDOR + ' SET modificado = ? WHERE fqdn = ?;', (MODIFICADO, FQDN)) 
209      except Exception as message: 
210          handler.log.error('no se puede modificar estado de servidor: %s', message) 
211          handler.log.exception(message) 
212          exit(1)

234  def Valida(): 
235      try: 
236          # comprobando que la db existe 
237          if not path.exists(SBCDB): 
238              raise ValueError("no existe la BD en " + path.abspath(SBCDB)) 
239           
240          # se conecta a la db 
241          conexion = connect(SBCDB, isolation_level=None) 
242          cursor=conexion.cursor() 
243           
244          # comprobando tabla configuracion 
245          for TABLA in (TABLA_CONFIGURACION, TABLA_SERVIDOR, TABLA_CARGAS, TABLA_OPERADOR, TABLA_ALERTA): 
246              cont = cursor.execute('pragma table_info(' + TABLA + ');').fetchall() 
247              if cont: handler.log.debug('la tabla %s parece estar bien', TABLA) 
248              else: 
249                  raise ValueError, "no existe la tabla " + TABLA       
250      except Exception as message: 
251          handler.log.error('ha ocurrido un problema al validar la integridad del modulo: %s', message) 
252          handler.log.exception(message) 
253          exit(1)
