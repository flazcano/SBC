10   from re import compile, findall

16   # funciones 
17   def Load(filename): # carga un archivo de configuracion 
18       try: configfile = open(filename, "r") 
19       except Exception as e: raise e 
20       try: configtext = configfile.read() 
21       except Exception as e: raise e 
22        
23       # Compile a pattern that matches our key-value line structure 
24       pattern = compile("\\n([\w_]+)[\t ]*([\w: \\\/~.-]+)") 
25        
26       # Find all matches to this pattern in the text of the config file 
27       tuples = findall(pattern, configtext) 
28        
29       # Create a new dictionary and fill it: for every tuple (key, value) in 
30       # the 'tuples' list, set ret[key] to value  
31       configuraciones = dict() 
32       for x in tuples: 
33           configuraciones[x[0]] = x[1] 
34        
35       # Return the fully-loaded dictionary object 
36       return configuraciones
