38   def EnviaCorreo(TOADDRESS, HOST, TIPO): 
39       handler.log.info('enviando alerta por correo a ' + TOADDRESS) 
40       try: 
41           mailServer = SMTP(SMTPHOST, SMTPPORT) 
42           mailServer.set_debuglevel(SMTPDEBUGLEVEL) 
43           mailServer.ehlo() 
44           mailServer.starttls() 
45           mailServer.ehlo() 
46           mailServer.login(SMTPUSER, SMTPPASS) 
47        
48           # creo el mensaje 
49           msg = MIMEMultipart('alternative') 
50           msg['From'] = SMTPUSER 
51           msg['To'] = TOADDRESS 
52           msg['Date'] = Utils.formatdate(localtime = 1) 
53           msg['Message-ID'] = Utils.make_msgid() 
54           msg['Subject'] = "Alerta SBC - " + HOST 
55           if TIPO == 0: 
56                
57               # Create the body of the message (a plain-text and an HTML version). 
58               text = "Estimado " + TOADDRESS + "!\n\n el servidor " + HOST + " ha vuelto a estar en linea" 
59               html = """\ 
60               <html> 
61                 <head></head> 
62                 <body> 
63                   <p>Estimado usuario!<br><br> 
64                      el servidor ha vuelto a estar en linea 
65                   </p> 
66                 </body> 
67               </html> 
68               """ 
69           elif TIPO == 1: 
70               # Create the body of the message (a plain-text and an HTML version). 
71               text = "Estimado " + TOADDRESS + "!\n\n el servidor " + HOST + " esta fuera de linea" 
72               html = """\ 
73               <html> 
74                 <head></head> 
75                 <body> 
76                   <p>Estimado usuario!<br><br> 
77                      el servidor esta fuera de linea 
78                   </p> 
79                 </body> 
80               </html> 
81               """ 
82           else: 
83               handler.log.info('el tipo ' + TIPO + ' no es un valor valido') 
84            
85           # Record the MIME types of both parts - text/plain and text/html. 
86           part1 = MIMEText(text, 'plain') 
87           part2 = MIMEText(html, 'html') 
88            
89           # Attach parts into message container. 
90           # According to RFC 2046, the last part of a multipart message, in this case 
91           # the HTML message, is best and preferred. 
92           msg.attach(part1) 
93           msg.attach(part2) 
94            
95           # envio del mensaje 
96           mailServer.sendmail(msg['From'], [msg['To']], msg.as_string()) 
97           mailServer.quit() 
98       except Exception as message: 
99           handler.log.error('error al enviar alerta por correo') 
100          handler.log.exception(message) 
101      finally: 
102          mailServer.close()

104  def EnviaJabber(TOJID, HOST, TIPO): 
105      handler.log.info('enviando alerta por Jabber a ' + TOJID) 
106      try: 
107          with warnings.catch_warnings(): 
108              warnings.simplefilter("ignore", category=DeprecationWarning) 
109              jid  = xmpp.protocol.JID(TOJID) 
110              jabber = xmpp.Client(XMPPCLIENT, debug=XMPPDEBUG) 
111              handler.log.debug('domain: ' + jid.getDomain()) 
112              connection = jabber.connect((XMPPHOST, XMPPPORT)) 
113              if not connection: 
114                  raise IOError('no se pudo conectar a ' + XMPPCLIENT) 
115              jabber.auth(XMPPUSER, XMPPPASS, XMPPRESOURCE) 
116              if not jabber.auth: 
117                  raise IOError('no se pudo autentificar con ' + XMPPCLIENT) 
118              jabber.sendInitPresence(requestRoster=1) 
119              if int(TIPO) == 0: 
120                  MSG = xmpp.Message(TOJID, 'El Servidor ' + HOST + ' se encuentra en linea') 
121                  MSG.setAttr('type', 'chat') 
122                  jabber.send(MSG) 
123              elif int(TIPO) == 1: 
124                  MSG = xmpp.Message(TOJID, 'El Servidor ' + HOST + ' se encuentra fuera de linea') 
125                  MSG.setAttr('type', 'chat') 
126                  jabber.send(MSG) 
127              else: 
128                  handler.log.error('el tipo ' + TIPO + ' no es un valor valido') 
129      except Exception as message: 
130          handler.log.error('error al enviar alerta por Jabber') 
131          handler.log.exception(message)
